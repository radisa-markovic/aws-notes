> **Prototype language**: custom blend of YAML (structure/metadata), TypeScript (types/interfaces), and Python (logic/indentation).
> **Stack**: React 19 · TypeScript · Vite · react-router-dom v7

## Syntax Legend

```
---                         # YAML metadata block delimiter
@prop                       # incoming prop declaration
@state                      # local component state
@context                    # consumed context value
@route(path)                # registers a page-level route
type Foo = { ... }          # TypeScript type (unchanged)
def methodName():           # Python-style method/handler block
    ...                     # indented body
=> expression               # arrow return (inline method shorthand)
[ ]                         # list / array
{ }                         # object / record
?                           # optional
|                           # union type
```

## Shared Types  `src/types/note.ts`

```ts
type Note = {
    id:        string
    title:     string
    body:      string
    createdAt: string        # ISO 8601
    updatedAt: string        # ISO 8601
    tags?:     string[]
    pinned?:   boolean
}

type NotesState = {
    notes:   Note[]
    loading: boolean
    error:   string | null
}

type NotesActions = {
    addNote:    (draft: Omit<Note, "id" | "createdAt" | "updatedAt">) -> void
    updateNote: (id: string, patch: Partial<Note>)                    -> void
    deleteNote: (id: string)                                          -> void
    pinNote:    (id: string)                                          -> void
}
```

## Context  `src/context/NotesContext.tsx`

```yaml
---
name: NotesContext
provides: NotesState & NotesActions
storage: localStorage["aws-notes"]
---
```

```python
@state notes:     Note[]    = load_from_storage() | []
@state loading:   boolean   = false
@state error:     string?   = null

def addNote(draft):
    note = {
        ...draft,
        id:        crypto.randomUUID(),
        createdAt: new Date().toISOString(),
        updatedAt: new Date().toISOString()
    }
    notes = [note, ...notes]
    persist(notes)

def updateNote(id, patch):
    notes = notes.map(n =>
        n.id == id ? { ...n, ...patch, updatedAt: now() } : n
    )
    persist(notes)

def deleteNote(id):
    notes = notes.filter(n => n.id != id)
    persist(notes)

def pinNote(id):
    updateNote(id, { pinned: !notes.find(n => n.id == id).pinned })
```

## Hook  `src/hooks/useNotes.ts`

```ts
def useNotes() -> NotesState & NotesActions:
    ctx = useContext(NotesContext)
    if not ctx: throw Error("useNotes must be inside <NotesProvider>")
    return ctx
```

## Component: `NoteForm`  `src/components/NoteForm.tsx`

```yaml
---
name: NoteForm
used-by: CreateNote, EditNote
---
```

```python
@prop initialValues: Partial<Note>  = {}
@prop onSubmit:      (data: Omit<Note, "id" | "createdAt" | "updatedAt">) -> void
@prop submitLabel:   string         = "Save"

@state title: string = initialValues.title | ""
@state body:  string = initialValues.body  | ""
@state tags:  string = initialValues.tags?.join(", ") | ""

def handleSubmit(e):
    e.preventDefault()
    onSubmit({
        title,
        body,
        tags: tags.split(",").map(t => t.trim()).filter(Boolean)
    })

ui:
    <form onSubmit=handleSubmit>
        <input    name="title"  value=title  onChange=setTitle  required />
        <textarea name="body"   value=body   onChange=setBody   required />
        <input    name="tags"   value=tags   onChange=setTags
                  placeholder="tag1, tag2" />
        <button type="submit"> submitLabel
```

## Page: `Home`  `src/pages/Home.tsx`

```yaml
---
route: "/"
title: All Notes
---
```

```python
@context { notes, deleteNote } = useNotes()

@state search: string = ""

def filteredNotes():
    q = search.toLowerCase()
    return notes
        .filter(n => n.title.toLowerCase().includes(q)
                  or n.body.toLowerCase().includes(q))
        .sort(by: pinned DESC, updatedAt DESC)

ui:
    <main>
        <input placeholder="Search notes…" value=search onChange=setSearch />
        if filteredNotes().length == 0:
            <p> "No notes yet. Create one!"
        else:
            <section class="note-grid">
                for note in filteredNotes():
                    <NoteCard note=note onDelete=deleteNote />
```
